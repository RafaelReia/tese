(define (fibs n)
  (let ((fibs
         (let loop ((previous 0)
                    (current 1)
                    (index 0))
           (if (= index n)
               (list)
               (cons current
                     (loop current
                           (+ previous current)
                           (+ index 1)))))))
    (for ([fib (in-list fibs)])
      (displayln fib))))
\end{lstlisting}
}
